	id = peerage.0
	desc = "EVTDESCpeerage.0"

# The estabilishment of the peerage
narrative_event = {
	id = peerage.1
	desc = "EVTDESCpeerage.1"
	title = "The Peerage of England"
	picture = "GFX_evt_emissary"
	border = "GFX_event_narrative_frame_diplomacy"
	
	is_triggered_only = yes
	
	trigger = {
	    OR = { 
		    AND = {
		        has_landed_title = k_england
	            NOT = { has_global_flag = peerage_initialised }
			}
		    AND = {
			    NOT = { has_landed_title = k_england }
	            NOT = { has_character_flag = peerage_initialised_message }
		    }
		}
	}
	
	option = {
		trigger = { NOT = { has_landed_title = k_england } }
	    name = "OK"
		set_character_flag = peerage_initialised_message
	}
	
	option = {
		trigger = { has_landed_title = k_england }
	    name = "This is the beginning of a new era"
		set_global_flag = peerage_initialised
        prestige = 250		
		hidden_tooltip = {
		    any_playable_ruler = {
			    limit = { ai = no }
				narrative_event = { id = peerage.1 }
			}
		}
	}
}
