			limit = { any_spouse = { character = FROM } }

					limit = {

			limit = {

						who = ROOT

						months = 60
					}
				}
			}
		}

		if = {

					opinion = {
						who = ROOT

				}
			}
		}
		FROM = {
			any_sibling = {
				limit = {
