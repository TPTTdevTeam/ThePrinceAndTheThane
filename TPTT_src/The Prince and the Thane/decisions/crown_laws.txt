				factor = 0

					trait = ghibelline

				}

				factor = 1.5

					trait = zealous

					trait = content
				}
			}	
		}
		ai_will_revoke = {
			factor = 0
		}

			hidden_tooltip = {
				revoke_law = investiture_law_0
				revoke_law = investiture_law_1
